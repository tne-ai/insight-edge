from tne.TNE import TNE

# Load data
df = session.get_object("handbags-shopping_list-no_run-1_1-with_periods_cleaned.csv")

# Filter data for Clutches
clutches_df = df[df['product_subgroup_desc'] == 'Clutch']

# Group by month and sum sales
monthly_sales = clutches_df.groupby('month')['sales'].sum().reset_index()

# Find the month with the highest sales (Best Time to Launch)
best_time_to_launch = monthly_sales.loc[monthly_sales['sales'].idxmax()]

# Create the result DataFrame
result = pd.DataFrame({
    'Best Time to Launch': [best_time_to_launch['month']],
    'Highest Sales': [best_time_to_launch['sales']]
})

print(result)
